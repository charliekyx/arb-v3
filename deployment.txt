# 替换为你的部署者私钥
export DEPLOYER_PRIVATE_KEY="<YOUR_DEPLOYER_PRIVATE_KEY>" 
# 替换为你的机器人钱包地址
export BOT_WALLET_ADDRESS="<YOUR_BOT_WALLET_ADDRESS>"
# Base 官方 RPC 或的本地节点
export RPC_URL="https://mainnet.base.org"

# 1. 部署合约
# 注意：合约文件名是 Contract.sol
forge create --rpc-url $RPC_URL \
  --private-key $DEPLOYER_PRIVATE_KEY \
  Contract.sol:FlashLoanExecutor

# 2. 将你的机器人钱包地址设置为合约的执行者 (Executor)
# 部署者将保持为 Owner
cast send <DEPLOYED_CONTRACT_ADDRESS> "setExecutor(address)" $BOT_WALLET_ADDRESS \
  --rpc-url $RPC_URL \
  --private-key $DEPLOYER_PRIVATE_KEY
